function sendInactiveEspionage(){
    let delay = 0;
    let step = 1000; 

    let inactives = document.querySelectorAll('#galaxytable tr.inactive_filter td.action a.espionage');

    if(inactives?.length > 0){
        inactives.forEach(i => {
            setTimeout( () => {
                i.click();
            }, delay);
            delay += step;
        });
    }

}
